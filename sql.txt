CREATE TABLE Livros( cod_livro INT NOT NULL AUTO_INCREMENT,
                      titulo_livro VARCHAR(50) NOT NULL, 
                      autor_livro VARCHAR(50) NOT NULL, 
                      cod_isbn CHAR(17) NOT NULL, 
                      nome_editora VARCHAR(50) NOT NULL, 
                      qtd_paginas INT NOT NULL, 
                      PRIMARY KEY (cod_livro))
ENGINE = InnoDB;
